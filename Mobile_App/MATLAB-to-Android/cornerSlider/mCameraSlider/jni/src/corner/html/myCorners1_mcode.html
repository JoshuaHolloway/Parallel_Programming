1function [R,G,B] = myCorners(R,G,B, fThresh, hThresh, mThresh) %#codegen
 2
 3%% Instantiate Corner Detection Objects
 4hHarris = vision.CornerDetector('Method','Harris corner detection (Harris & Stephens)', ...
 5  'MaximumCornerCount', 10000, 'CornerThreshold', eps, 'CornerLocationOutputPort', false, ...
 6  'MetricMatrixOutputPort', true);
 7
 8hFAST = vision.CornerDetector('Method','Local intensity comparison (Rosten & Drummond)', ...
 9    'MaximumCornerCount', 10000, 'CornerThreshold', eps, 'CornerLocationOutputPort', false, ...
10    'MetricMatrixOutputPort', true);
11
12hminEigen = vision.CornerDetector('Method','Minimum eigenvalue (Shi & Tomasi)', ...
13    'MaximumCornerCount', 10000, 'CornerThreshold', eps, 'CornerLocationOutputPort', false, ...
14    'MetricMatrixOutputPort', true);
15
16hFindLocalMax = vision.LocalMaximaFinder('MaximumNumLocalMaxima', 1000, ...
17    'NeighborhoodSize', [11 11], 'ThresholdSource', 'Input port');
18
19hMarkerInserter = vision.MarkerInserter('Shape','Plus','Size',3,'BorderColor','White');
20  
21% hGrayScaleConverter = vision.ColorSpaceConverter('Conversion', 'RGB to intensity');
22
23%% Concatenate color planes and convert to single precision
24%  Then convert to grayscale
25% I = single(cat(3,R,G,B));
26% IGray = step(hGrayScaleConverter, I);
27
28% for codegen
29coef = single([0.2989360 0.5870430 0.1140209]);
30IGray = single(R)*coef(1) + single(G)*coef(2) + single(B)*coef(3);
31
32
33%% FAST (Rosten and Drummond) Corners for higher performance
34minThresh = single(200);
35if (fThresh < minThresh), fThresh = minThresh; end
36metric = step(hFAST,IGray);
37fastPoints = step(hFindLocalMax, metric, fThresh);
38%R = insertPlus(R,fastPoints);
39R = step(hMarkerInserter,R,fastPoints);
40%I = insertMarker(R, fastPoints,'+', 'Color', [255 0 0]);
41
42%% Classical Harris Corners
43minThresh = single(10000);
44if (hThresh < minThresh), hThresh = minThresh; end
45metric = step(hHarris,IGray);
46harrisPoints = step(hFindLocalMax, metric, hThresh);
47% G = insertPlus(G,harrisPoints);
48G = step(hMarkerInserter,G,harrisPoints);
49% I2 = insertMarker(G, harrisPoints,'+', 'Color', [0 255 0]);
50
51%% Shi and Tomasi Minimum Eigenvalue Corners
52minThresh = single(1000);
53if (mThresh < minThresh), mThresh = minThresh; end
54metric = step(hminEigen,IGray);
55minEigenPoints = step(hFindLocalMax, metric, mThresh);
56% B = insertPlus(B,minEigenPoints);
57B = step(hMarkerInserter,B,minEigenPoints);
58% I3 = insertMarker(B, minEigenPoints,'+', 'Color', [0 0 255]);
59
60% R = uint8(I(:,:,1));
61% G = uint8(I2(:,:,2));
62% B = uint8(I3(:,:,3));
